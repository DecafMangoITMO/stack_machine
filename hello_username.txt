section .data:
    hello: "hello"
    username: bf 30
    pointer: username
section .text:
    lit -1
    .loop1:
        di
        lit 0
        cmp
        drop
        jz .next
        ei
        jmp .loop1

    .next:
        drop
        lit hello
        lit pointer
        pop
        call .print_str
        call .print_space
        lit username
        lit pointer
        pop
        call .print_str
        halt

    .print_str:
        .loop2:
            lit pointer
            push
            push
            lit 0
            cmp
            jz .stop2
            drop 
            lit out 
            pop
            lit pointer
            dup
            push
            inc
            switch
            pop
            jmp .loop2
        .stop2:
            ret

    .print_space:
        lit 32
        lit out
        pop
        ret
    
    .int1:
        drop
        lit in
        push
        dup
        lit pointer
        push
        pop
        lit pointer
        dup
        push
        inc
        switch
        pop
        iret
